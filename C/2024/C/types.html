Complément sur les types

1. Type enum

Exemple
// définition d'un type enum bool
enum bool { false, true } ;

// définition d'un type enum jour
enum jour { dimanche,lundi,mardi,mercredi,jeudi,vendredi,samedi } ;

// déclaration de 2 variables  aujourdhui et demain
enum jour aujourdhui, demain ;

aujourdhui = lundi ;
demain = mardi ;

2. Type struct

Exemple
struct person
{
  char firstname[20], lastname[20] ;
  int age ;
} ;

int main()
{
  struct person moi = { "Jacques","Dupont",20 } ;
  struct person toi, *lui ;

  strcpy(toi.firstname,"Anne") ;
  strcpy(toi.lastname,"Dubois") ;
  toi.age = 25 ;

  lui = malloc(sizeof(struct person)) ;
  if (lui==NULL) perror("lui") ;

  scanf("%s %s %d",lui->firstname,lui->lastname,&lui->age) ;

  printf("%s %s %d ans\n",moi.firstname,moi.lastname,moi.age) ;
  printf("%s %s %d ans\n",toi.firstname,toi.lastname,toi.age) ;
  printf("%s %s %d ans\n",lui->firstname,lui->lastname,lui->age) ;

  free(lui) ;

  return 0 ;
}

3. Type union

Exemple
struct FrAddress  // format d'une adresse française
{
  int numero ;
  char rue[20] ;
  int postal_code  ;
  char ville[10] ;
} ;

struct USAddress  // format d'une adresse américaine
{
  int house_numb;
  char street[15], direction[5];
  char locality[10], province[6], postal_code[8];
} ;

struct UKAddress  // format d'une adresse anglaise
{
  int house_numb ;
  char street[15] ;
  char town[10], region[10], postal_code[8] ;
} ;

union Address  // adresse générique
{
  enum { uk,fr,us } pays ;   // le champ pays indique le type, parmi les 3 possibles

  struct UKAddress uk ;
  struct FrAddress fr ;
  struct USAddress us ;
} ;

union Address ici = {   // la variable ici est initialisée avec l'adresse de la fac
                      .pays=fr,
                      .fr.numero=2 ,
                      .fr.rue="bd Lavoisier",
                      .fr.postal_code=49000,
                      .fr.ville="Angers"
} ;

4. typedef

Exemples
// définition de quelques types

typedef unsigned char Bool ;
typedef char String[255] ;
typedef int *Ptr ;
typedef void procedure(void) ;

// utilisation des ces types

Bool done ;     // <=> unsigned char done ;
String s[10] ; 	// <=> char s[255][10] ;
Ptr ptr ;       // <=> int *ptr ;
procedure f ;   // <=> void f(void) ;
