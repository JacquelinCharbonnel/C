Fichiers de données

Manipuler un fichier de données se fait via le type FILE *, et des fonctions de la bibliothèque standard (déclarées dans stdio.h) :

  • d’ouverture : fopen

  • de fermeture : fclose

  • d’écriture : fprintf, fputs

  • de lecture : fscanf, fgets

Exemple
#include <stdio.h>

#define ORIGINAL "monfichier"
#define COPIE "/tmp/mondouble"
#define L 60

int main()
{
  FILE *in, *out ;
  char s[L] ;
  int nl = 0 ;

  in = fopen(ORIGINAL,"r") ;
  if (in==NULL) { perror(ORIGINAL) ; return 1 ; }
  out = fopen(COPIE,"w") ;
  if (out==NULL) { perror(COPIE) ; return 2 ; }

  while(fgets(s,L-1,in)!=NULL)
  {
    fprintf(out,"%02d: %s",nl,s) ;
    nl++ ;
  }

  fclose(in) ; fclose(out) ;
  return 0 ;
}
Pour voir les fonctions de manipulation de fichiers, faire man stdio.
